to go
clean
fillscreen 70 3
setcolor 20
stemstar true
end

to stemstar :smaller
repeat 8
 [storestate
  stem
  if :smaller 
    [setscale .3 
     setcolor 18 setshade 90 
     stemstar false]
  restorestate
  rt 45]
end

to stem
make "i 0
setpensize 20
seg 98 102 .01
seg 148 152 -.015
seg 460 440 .01
end

to seg :n1 :n2 :d
repeat random2 :n1 :n2
 [fd .5
  rt :i 
  make "i :i + :d
  setpensize pensize - .025]
end
